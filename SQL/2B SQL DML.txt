-- RESET OLD TABLE (MySQL does not support PURGE or PL/SQL blocks)
DROP TABLE IF EXISTS STUDENT;

-- CREATE NEW STUDENT TABLE
CREATE TABLE STUDENT (
  roll_no INT,
  name VARCHAR(30),
  marks INT,
  dept VARCHAR(20)
);

-- INSERT SAMPLE DATA
INSERT INTO STUDENT VALUES (1,'Rohit',85,'IT');
INSERT INTO STUDENT VALUES (2,'Sneha',72,'HR');
INSERT INTO STUDENT VALUES (3,'Amit',90,'IT');
INSERT INTO STUDENT VALUES (4,'Meera',60,'FINANCE');
INSERT INTO STUDENT VALUES (5,'Ankit',55,'HR');

-- 1) SELECT *
SELECT * FROM STUDENT;

-- 2) SELECT with WHERE
SELECT name, marks FROM STUDENT WHERE marks > 70;

-- 3) LIKE operator
SELECT * FROM STUDENT WHERE name LIKE 'A%';

-- 4) BETWEEN operator
SELECT * FROM STUDENT WHERE marks BETWEEN 60 AND 90;

-- 5) Function (UPPER)
SELECT name, UPPER(dept) AS DEPARTMENT FROM STUDENT;

-- 6) UPDATE
UPDATE STUDENT SET marks = marks + 5 WHERE dept = 'IT';

-- 7) DELETE
DELETE FROM STUDENT WHERE marks < 60;

-- 8) UNION
SELECT name FROM STUDENT WHERE dept='IT'
UNION
SELECT name FROM STUDENT WHERE dept='HR';

-- 9) MINUS equivalent for MySQL (MySQL does NOT support MINUS)
SELECT s1.roll_no
FROM STUDENT s1
LEFT JOIN STUDENT s2
     ON s1.roll_no = s2.roll_no AND s2.dept = 'IT'
WHERE s2.roll_no IS NULL;

-- 10) ORDER BY
SELECT * FROM STUDENT ORDER BY marks DESC;
